-- Table Queries 

CREATE TABLE IF NOT EXISTS public.hr_tbl_evaluation_setup_grades
(
    evaluation_setup_grade_id integer NOT NULL DEFAULT nextval('hr_tbl_evaluation_setup_grades_seq'::regclass),
    grade character varying(100) COLLATE pg_catalog."default" NOT NULL,
    initial_marks double precision NOT NULL,
    final_marks double precision NOT NULL,
    description character varying(255) COLLATE pg_catalog."default",
    ser_evaluation_setup_id integer NOT NULL,
    CONSTRAINT hr_tbl_evaluation_setup_grades_pkey PRIMARY KEY (evaluation_setup_grade_id),
    CONSTRAINT fk_evaluation_setup FOREIGN KEY (ser_evaluation_setup_id)
        REFERENCES public.hr_tbl_evaluation_setup (ser_evaluation_setup_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.hr_tbl_evaluation_setup_grades
    OWNER to postgres;
